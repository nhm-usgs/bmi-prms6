# The BMI spec is "baked in" to the bmi-prms shared object.
add_executable(run_${bmiprms_lib} main.f90)
target_link_libraries(run_bmiprms6 ${EXTRA_LIBS})

add_library(bmiprms6 SHARED m_prms.f90 bmi_prms.f90 bmi.f90)
target_link_libraries(${bmiprms_lib} ${EXTRA_LIBS})

install(
  PROGRAMS ${CMAKE_CURRENT_BINARY_DIR}/run_${bmiprms_lib}
  DESTINATION bin)
install(
  TARGETS ${bmiprms_lib}
  DESTINATION lib)
install(
  FILES ${CMAKE_CURRENT_BINARY_DIR}/bmi.mod
        ${CMAKE_CURRENT_BINARY_DIR}/${bmiprms_lib}.mod
  DESTINATION include)
